-- 1. CREACIÓN DE LA BASE DE DATOS
CREATE DATABASE IF NOT EXISTS avance_proyecto;
USE avance_proyecto;

-- ==========================================
-- 2. TABLA PRINCIPAL DE USUARIOS
-- ==========================================
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    rol ENUM('admin', 'fan') DEFAULT 'fan',
    fecha_nacimiento DATE,
    sexo VARCHAR(50) DEFAULT 'Prefiero no decirlo',
    descripcion TEXT,
    privacidad ENUM('publico', 'privado') DEFAULT 'publico',
    foto_perfil VARCHAR(500) DEFAULT 'https://i.imgur.com/3p3E53e.png',
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- 3. TABLAS DE CONTENIDO PRINCIPAL
-- ==========================================
CREATE TABLE IF NOT EXISTS canciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    artista VARCHAR(150) NOT NULL,
    genero VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS tareas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    descripcion TEXT,
    prioridad VARCHAR(50),
    completada BOOLEAN DEFAULT FALSE
);

CREATE TABLE IF NOT EXISTS eventos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150) NOT NULL,
    fecha DATETIME NOT NULL,
    lugar VARCHAR(150) NOT NULL
);

-- ==========================================
-- 4. TABLAS DE INTERACCIÓN Y COMUNIDAD
-- ==========================================
CREATE TABLE IF NOT EXISTS foros (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(200) NOT NULL,
    descripcion TEXT NOT NULL,
    estado ENUM('pendiente', 'aprobado', 'rechazado') DEFAULT 'pendiente',
    user_id INT NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS sugerencias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    descripcion TEXT NOT NULL,
    imagen_url VARCHAR(500),
    user_id INT NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS comentarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    entidad_id INT NOT NULL,
    tipo_entidad VARCHAR(50) NOT NULL, -- 'cancion', 'tarea', 'foro'
    comentario TEXT NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS calificaciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    entidad_id INT NOT NULL,
    tipo_entidad VARCHAR(50) NOT NULL, -- 'cancion', 'tarea'
    puntuacion DECIMAL(3,1) NOT NULL,
    UNIQUE KEY (user_id, entidad_id, tipo_entidad),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- ==========================================
-- 5. TABLAS DE RED SOCIAL Y CHAT
-- ==========================================
CREATE TABLE IF NOT EXISTS seguidores (
    seguidor_id INT NOT NULL,
    seguido_id INT NOT NULL,
    fecha_seguimiento TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (seguidor_id, seguido_id),
    FOREIGN KEY (seguidor_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (seguido_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS mensajes_directos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    emisor_id INT NOT NULL,
    receptor_id INT NOT NULL,
    mensaje TEXT NOT NULL,
    leido BOOLEAN DEFAULT FALSE,
    fecha_envio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (emisor_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (receptor_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS notificaciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    tipo VARCHAR(50) NOT NULL, -- 'aviso', 'seguidor', 'mensaje'
    mensaje TEXT NOT NULL,
    remitente_username VARCHAR(100),
    leido BOOLEAN DEFAULT FALSE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES users(id) ON DELETE CASCADE
);

-- ==========================================
-- 6. TABLAS DE PANTALLA DE INICIO (HOME)
-- ==========================================
CREATE TABLE IF NOT EXISTS avisos_globales (
    id INT AUTO_INCREMENT PRIMARY KEY,
    mensaje TEXT NOT NULL,
    autor VARCHAR(100) NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS banners (
    posicion VARCHAR(10) PRIMARY KEY, -- 'top' o 'bottom'
    imagen_url VARCHAR(500)
);

CREATE TABLE IF NOT EXISTS destacados_home (
    slot VARCHAR(20) PRIMARY KEY, -- 'cancion1', 'album1', etc.
    item_id INT
);

CREATE TABLE IF NOT EXISTS encuestas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    pregunta VARCHAR(255) NOT NULL,
    opcion1 VARCHAR(100) NOT NULL,
    opcion2 VARCHAR(100) NOT NULL,
    opcion3 VARCHAR(100),
    opcion4 VARCHAR(100),
    activa BOOLEAN DEFAULT TRUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS encuesta_votos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    encuesta_id INT NOT NULL,
    user_id INT NOT NULL,
    opcion_seleccionada INT NOT NULL,
    UNIQUE KEY (encuesta_id, user_id),
    FOREIGN KEY (encuesta_id) REFERENCES encuestas(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
